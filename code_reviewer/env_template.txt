# Azure OpenAI Configuration for Multi-Agent Code Review System
# Copy this content to a .env file and fill in your actual values

# =============================================================================
# Azure AD Authentication (Required)
# These values are needed for client secret (application password) authentication
# =============================================================================

AZURE_TENANT_ID=your-tenant-id-here
AZURE_CLIENT_ID=your-application-client-id-here
AZURE_CLIENT_SECRET=your-client-secret-application-password-here

# =============================================================================
# Azure OpenAI Service Configuration (Required)
# =============================================================================

AZURE_ENDPOINT=https://your-resource.openai.azure.com

# =============================================================================
# Optional Configuration
# =============================================================================

# Deployment name for your GPT model (default: gpt-4)
AZURE_DEPLOYMENT_NAME=gpt-4

# API version (default: 2024-02-15-preview)
AZURE_API_VERSION=2024-02-15-preview

# Maximum tokens per request (default: 2000)
AZURE_MAX_TOKENS=2000

# Temperature/creativity level (default: 0.1)
AZURE_TEMPERATURE=0.1

# =============================================================================
# Where to find these values in Azure Portal:
# =============================================================================
# 1. AZURE_TENANT_ID: Azure AD > Properties > Tenant ID
# 2. AZURE_CLIENT_ID: App registrations > Your app > Application ID  
# 3. AZURE_CLIENT_SECRET: App registrations > Your app > Certificates & secrets > Client secrets > Value
#    (This is the secret string that proves your app's identity - the application password)
# 4. AZURE_ENDPOINT: Azure OpenAI > Keys and Endpoint > Endpoint
# 5. AZURE_DEPLOYMENT_NAME: Azure OpenAI > Deployments > Your deployment name
